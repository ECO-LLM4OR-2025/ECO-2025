+++++
Sets:
  AnimalTypes = {cows, sheep, chickens}

Variables:
  x_cows ∈ Z+, number of cows raised
  x_sheep ∈ Z+, number of sheep raised
  x_chickens ∈ Z+, number of chickens raised

Parameters:
  p_cows = 500   selling price per cow
  c_cows = 100   feed cost per cow
  p_sheep = 200  selling price per sheep
  c_sheep = 80   feed cost per sheep
  p_chick = 8    selling price per chicken
  c_chick = 5    feed cost per chicken
  m_cows = 10    manure units per cow per day
  m_sheep = 5    manure units per sheep per day
  m_chick = 3    manure units per chicken per day
  M_max = 800    maximum manure units handled per day
  Chick_max = 50 maximum chickens
  Cow_min = 10   minimum cows
  Sheep_min = 20 minimum sheep
  N_max = 100    maximum total animals

Derived parameters:
  π_cows = p_cows - c_cows = 400  profit per cow
  π_sheep = p_sheep - c_sheep = 120  profit per sheep
  π_chick = p_chick - c_chick = 3    profit per chicken

Constraints:
  1. Manure capacity: 10 x_cows + 5 x_sheep + 3 x_chickens ≤ 800
  2. Chicken capacity:       x_chickens ≤ 50
  3. Minimum cows:           x_cows ≥ 10
  4. Minimum sheep:          x_sheep ≥ 20
  5. Total animals:          x_cows + x_sheep + x_chickens ≤ 100
  6. Integrality & nonnegativity: x_* ∈ {0,1,2,...}

Objective:
  Maximize Z = 400 x_cows + 120 x_sheep + 3 x_chickens
+++++

=====
import gurobipy as gp
from gurobipy import GRB

# Initialize model
model = gp.Model("farm_profit_maximization")

# Decision variables
cows    = model.addVar(lb=0, vtype=GRB.INTEGER, name="cows")
sheep   = model.addVar(lb=0, vtype=GRB.INTEGER, name="sheep")
chickens= model.addVar(lb=0, vtype=GRB.INTEGER, name="chickens")

# Parameters (profits)
profit_cow   = 500 - 100   # 400
profit_sheep = 200 - 80    # 120
profit_chick = 8   - 5     # 3

# Constraints
# 1. Manure capacity
model.addConstr(10*cows + 5*sheep + 3*chickens <= 800, name="manure_cap")
# 2. Chicken capacity
model.addConstr(chickens <= 50, name="chicken_cap")
# 3. Minimum cows
model.addConstr(cows >= 10, name="min_cows")
# 4. Minimum sheep
model.addConstr(sheep >= 20, name="min_sheep")
# 5. Total animals
model.addConstr(cows + sheep + chickens <= 100, name="total_animals")

# Objective: maximize profit
model.setObjective(profit_cow*cows + profit_sheep*sheep + profit_chick*chickens,
                   GRB.MAXIMIZE)

# Solve
model.optimize()

# Write optimal value to file
if model.status == GRB.OPTIMAL:
    with open('ref_optimal_value.txt', 'w') as f:
        f.write(str(model.objVal))
else:
    with open('ref_optimal_value.txt', 'w') as f:
        f.write("No optimal solution found.")
