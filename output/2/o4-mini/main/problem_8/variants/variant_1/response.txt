+++++
Sets:
Products = {Scooter, Bike}

Variables:
x_Scooter ∈ Integer, ≥ 0  (number of scooters to produce)
x_Bike    ∈ Integer, ≥ 0  (number of bikes to produce)

Parameters:
p_Scooter = 200      (profit per scooter)
p_Bike    = 300      (profit per bike)
d_Scooter = 2        (design hours per scooter)
d_Bike    = 4        (design hours per bike)
e_Scooter = 3        (engineering hours per scooter)
e_Bike    = 5        (engineering hours per bike)
D_capacity = 5000    (total design hours available)
E_capacity = 6000    (total engineering hours available)

Constraints:
1) Design capacity:  d_Scooter * x_Scooter + d_Bike * x_Bike ≤ D_capacity
   ⇒ 2·x_Scooter + 4·x_Bike ≤ 5000
2) Engineering capacity:  e_Scooter * x_Scooter + e_Bike * x_Bike ≤ E_capacity
   ⇒ 3·x_Scooter + 5·x_Bike ≤ 6000
3) Nonnegativity and integrality: x_Scooter, x_Bike ≥ 0 and integer

Objective:
Maximize total profit = p_Scooter * x_Scooter + p_Bike * x_Bike
                         = 200·x_Scooter + 300·x_Bike
+++++

=====
import gurobipy as gp
from gurobipy import GRB

# Create the optimization model
model = gp.Model("scooter_bike_production")

# Decision variables: number of scooters and bikes to produce
x_scooter = model.addVar(vtype=GRB.INTEGER, name="x_scooter", lb=0)
x_bike    = model.addVar(vtype=GRB.INTEGER, name="x_bike", lb=0)

# Parameters
profit_scooter = 200
profit_bike    = 300
design_scooter = 2
design_bike    = 4
eng_scooter    = 3
eng_bike       = 5
design_cap     = 5000
eng_cap        = 6000

# Objective: maximize total profit
model.setObjective(
    profit_scooter * x_scooter + profit_bike * x_bike,
    GRB.MAXIMIZE
)

# Constraints
model.addConstr(design_scooter * x_scooter + design_bike * x_bike <= design_cap,
                name="design_capacity")
model.addConstr(eng_scooter * x_scooter + eng_bike * x_bike <= eng_cap,
                name="engineering_capacity")

# Optimize the model
model.optimize()

# Write the optimal objective value to a file
if model.status == GRB.OPTIMAL:
    with open('ref_optimal_value.txt', 'w') as f:
        f.write(str(model.objVal))
