Sets:
  P = {chair, dresser}

Variables:
  x[p] : number of units of product p to produce, continuous ≥ 0, for p in P

Parameters:
  profit[p] : profit per unit of product p
    profit[chair] = 43
    profit[dresser] = 52
  stain_req[p] : gallons of stain required per unit of product p
    stain_req[chair] = 1.4
    stain_req[dresser] = 1.1
  wood_req[p] : lengths of oak wood required per unit of product p
    wood_req[chair] = 2
    wood_req[dresser] = 3
  stain_avail = 17    # total gallons of stain available per week
  wood_avail = 11     # total lengths of oak wood available per week

Constraints:
  1) Stain availability: sum_p stain_req[p] * x[p] ≤ stain_avail
  2) Wood availability:  sum_p wood_req[p]  * x[p] ≤ wood_avail
  3) Nonnegativity:     x[p] ≥ 0 for all p in P

Objective:
  Maximize total profit: ∑_{p ∈ P} profit[p] * x[p]
