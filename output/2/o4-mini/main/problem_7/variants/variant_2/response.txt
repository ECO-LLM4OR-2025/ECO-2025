+++++
Sets:
Products: P = {chair, dresser}

Variables:
x_chair ∈ ℤ≥0 (number of chairs to produce)
x_dresser ∈ ℤ≥0 (number of dressers to produce)

Parameters:
profit_chair = 43       (profit per chair)
profit_dresser = 52     (profit per dresser)
stain_chair = 1.4       (gallons of stain per chair)
stain_dresser = 1.1     (gallons of stain per dresser)
wood_chair = 2          (lengths of oak wood per chair)
wood_dresser = 3        (lengths of oak wood per dresser)
avail_stain = 17        (total gallons of stain available per week)
avail_wood = 11         (total lengths of oak wood available per week)

Constraints:
1) Stain constraint: stain_chair * x_chair + stain_dresser * x_dresser ≤ avail_stain
2) Wood constraint: wood_chair * x_chair + wood_dresser * x_dresser ≤ avail_wood
3) Non-negativity and integrality: x_chair ≥ 0, x_dresser ≥ 0 and integer

Objective:
Maximize total profit = profit_chair * x_chair + profit_dresser * x_dresser
+++++

=====
import gurobipy as gp
from gurobipy import GRB

# Create model
model = gp.Model("Elm_Furniture_Production")

# Decision variables
x_chair = model.addVar(vtype=GRB.INTEGER, name="x_chair", lb=0)
x_dresser = model.addVar(vtype=GRB.INTEGER, name="x_dresser", lb=0)

# Parameters
profit_chair = 43
profit_dresser = 52
stain_chair = 1.4
stain_dresser = 1.1
wood_chair = 2
wood_dresser = 3
avail_stain = 17
avail_wood = 11

# Constraints
model.addConstr(stain_chair * x_chair + stain_dresser * x_dresser <= avail_stain, 
                name="StainLimit")
model.addConstr(wood_chair * x_chair + wood_dresser * x_dresser <= avail_wood, 
                name="WoodLimit")

# Objective
model.setObjective(profit_chair * x_chair + profit_dresser * x_dresser, GRB.MAXIMIZE)

# Optimize
model.optimize()

# Save optimal objective value
if model.Status == GRB.OPTIMAL:
    optimal_value = model.ObjVal
    with open('ref_optimal_value.txt', 'w') as f:
        f.write(str(optimal_value))
    # Optionally print the solution
    # print(f"Optimal production: chairs={x_chair.X}, dressers={x_dresser.X}, profit={optimal_value}")
