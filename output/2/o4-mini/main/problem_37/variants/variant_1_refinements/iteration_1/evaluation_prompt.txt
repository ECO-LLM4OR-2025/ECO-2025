import gurobipy as gp
from gurobipy import GRB

# Create model
model = gp.Model("terracotta_jars")

# Parameters

baking_thin = 90

shaping_capacity = 3000
baking_capacity = 4000
profit_thin = 5
profit_stubby = 9

# Variables
x_thin = model.addVar(vtype=GRB.INTEGER, name="x_thin", lb=0)
